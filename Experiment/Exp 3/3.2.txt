-- AGGREGATE FUNCTION -> SUM(),MIN(),MAX(),AVG(),COUNT()

-- (1)
-- COUNT THE NUMBER OF EMPLOYEES IN Each CITY

-- (I)
Select emp_city,count(*) AS CNT from Employees
group by emp_city

-- (II)

Select emp_city,count(EMP_ID) AS CNT from Employees
group by emp_city


-- COUNT THE NUMBER OF EMPLOYEE IN EACH CITY
-- WHOSE SALARY IS >=90000 AND SORT ON THE BASIS OF COUNT IN DESC AND IF CNT ARE SAME THE SORT ON THE
-- BASIS OF EMP_CITY IN DESC ORDER

SELECT EMP_CITY,SUM(CASE WHEN EMP_SALARY>=90000 THEN 1 ELSE 0 END) AS CNT
FROM EMPLOYEES
GROUP BY EMP_CITY
ORDER BY CNT DESC,EMP_CITY DESC

-- FIND THE AVERAGE EMPLOYEE SALARY IN EACH CITY

SELECT EMP_CITY,AVG(EMP_SALARY)::NUMERIC(20,2) AS AVGS
FROM EMPLOYEES
GROUP BY EMP_CITY
-- 
-- FIND THE EMP_CITY  WHOSE AVG SALARY IS GRETAER THEN 70000

SELECT EMP_CITY,AVG(EMP_SALARY)::NUMERIC(20,2) AS AVGS
FROM EMPLOYEES
GROUP BY EMP_CITY
HAVING AVG(EMP_SALARY)>70000
